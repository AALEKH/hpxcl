247  global_work_offset[i] = dim[i].offset;
248  global_work_size[i] = dim[i].size;
249  local_work_size_[i] = dim[i].local_size;

251

253  for(size_t i = 0; i < DIM; i++)

257  local_work_size = local_work_size_;

260  }

263  return enqueue(DIM, global_work_offset, global_work_size,
264  local_work_size, events);
